*** elpy
    #+begin_src emacs-lisp
      (use-package elpy
	:ensure t
	:init
	(elpy-enable)
	;; start python shells in directory of python file
	(setq elpy-shell-starting-directory 'current-directory)
	;; display the Python shell after sending something to it
	; (setq elpy-shell-display-buffer-after-send '1)
	)

      ;; fix issue when running C-c C-c
      (setq python-shell-completion-native-disabled-interpreters '("python3"))

      ;; attaches a dedicated python shell to the current python buffer
      (add-hook 'elpy-mode-hook (lambda () (elpy-shell-toggle-dedicated-shell 1)))

      ;; use flycheck instead of flymake
      (when (load "flycheck" t t)
	(setq elpy-modules (delq 'elpy-module-flymake elpy-modules))
	(add-hook 'elpy-mode-hook 'flycheck-mode))
    #+end_src
